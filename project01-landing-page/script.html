const typingText = document.getElementById('typing-text'); const phrases = ["Fullstack Developer.", "UI/UX Enthusiast.", "Problem Solver.", "Amelia."]; let phraseIndex = 0; let characterIndex = 0; let isDeleting = false; let typeSpeed = 150; function typeEffect() { const currentPhrase = phrases[phraseIndex]; if (isDeleting) { typingText.textContent = currentPhrase.substring(0, characterIndex - 1); characterIndex--; typeSpeed = 80; } else { typingText.textContent = currentPhrase.substring(0, characterIndex + 1); characterIndex++; typeSpeed = 150; } if (!isDeleting && characterIndex === currentPhrase.length) { isDeleting = true; typeSpeed = 2000; // Jeda saat kata selesai } else if (isDeleting && characterIndex === 0) { isDeleting = false; phraseIndex = (phraseIndex + 1) % phrases.length; typeSpeed = 500; } setTimeout(typeEffect, typeSpeed); } // Jalankan saat halaman dimuat document.addEventListener('DOMContentLoaded', () => { typeEffect(); });
